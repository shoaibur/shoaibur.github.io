Linear Regression


import numpy as np

class LinearRegression:
    def __init__(self, lr=0.1, it=100):
        self.lr = lr
        self.it = it
        self.costs = []
    
    def train(self, Xtr, ytr, Xval=None, yval=None):
        m, n = Xtr.shape
        W = np.random.rand(n, 1)
        b = 0
        for _ in range(self.it):
            y_hat = np.dot(Xtr, W) + b
            cost = (1/m) * np.sum((y_hat - ytr) ** 2)
            self.costs.append(cost)
            dw = (-2/m) * np.dot(Xtr.T, (ytr - y_hat))
            db = (-2/m) * np.sum(ytr - y_hat)
            W -= self.lr * dw
            b -= self.lr * db
        self.W = W
        self.b = b
    
    def predict(self, Xts):
        return np.dot(Xts, self.W) + self.b

# Example usage
if __name__ == "__main__":
    Xtr = np.random.rand(15, 3)
    ytr = np.random.rand(15, 1)
    Xts = np.random.rand(5, 3)
    
    model = LinearRegression(lr=0.01, it=1000)
    model.train(Xtr, ytr)
    predictions = model.predict(Xts)
    print("Predictions:", predictions)
